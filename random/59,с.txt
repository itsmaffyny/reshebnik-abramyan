#include <stdio.h>
#include <stdlib.h>
#include <string.h>

// Task: Dynamic Array of Strings
// C does not have a built-in type for an array of strings. 
// An array of strings is just a dynamic array that stores pointers, 
// and each pointer points to its own separate block of memory on the heap. 
// You need to build it, expand it, and most importantly, destroy it securely.

int main (void){
char passwords[] = "pasw1 pasw2 pasw3 idk";
int i=1; for (i=1; passwords[i] != '\0'; i++){}
printf("%d", i); //22 

char *pnt_temp=malloc(22* sizeof(char));
memcpy(pnt_temp, passwords, 22* sizeof(char));

char *check =malloc(17* sizeof(char));
if (check !=NULL){
    memcpy(check,  pnt_temp, 16* sizeof(char));
    check[16]='\0';
    memset(pnt_temp, 0, 22* sizeof(char));
    free(pnt_temp);
}
pnt_temp=check;
for(int i=0; i<16; i++){
    printf("%c",pnt_temp[i]);
}
free(pnt_temp);
return 0;
}
